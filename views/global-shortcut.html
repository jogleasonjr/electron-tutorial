global-shortcut module

main.js

global-shortcut is a main process module that can register/unregister a global keyboard shortcut with the operating system so that you can customize the operations for various shortcuts.

var app = require('app');
var globalShortcut = require('global-shortcut');
app.on('ready', function() {
  globalShortcut.register('ctrl+x', function() {
    console.log('ctrl+x is pressed');
  });
});
Methods
  • globalShortcut.register(accelerator, callback)
  • globalShortcut.isRegistered(accelerator)
  • globalShortcut.unregister(accelerator)
  • globalShortcut.unregisterAll()
