dialog module

main.js

dialog is a main process api for showing native system dialogs, such as opening files or alerting, so web applications can deliver the same user experience as native applications.

var dialog = require('dialog');
dialog.showOpenDialog({
  title: 'Open Text Files',
  properties: ['openFile', 'multiSelections'],
  filters: [
    { name: 'Text Files', extensions: ['txt'] },
    { name: 'All Files', extensions: ['*'] }
  ]
}, function (files) {
  console.log(require('util').inspect(files));
});
