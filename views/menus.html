menu module

A main process api for showing native system dialogs, such as opening files or alerting, so web applications can deliver the same user experience as native applications.

example
var path = require('path');
var Tray = require('tray');
var Menu = require('menu');
var template = [
  {
    label: 'Surly Darkness',
    click: function (item, focusedWindow) {
      console.log('Surly it is!');
    }
  },
  {
    label: 'More Surly Darkness',
    click: function (item, focusedWindow) {
      console.log('Yay!');
    }
  }
];
var menu = Menu.buildFromTemplate(template);
var tray = new Tray(path.join(__dirname, '../images/beer.png'));
tray.setToolTip('Buy me a beer!');
tray.setContextMenu(menu);
